# find-digit-
n=int(input("Enter the digit from 0 to 4:"))
print("Entered digit id:",end='')
if n==0:
    print("Zero")
elif n==1:
    print("One")
elif n==2:
    print("Two")
elif n==3:
    print("Three")
elif n==4:
    print("Four")
else:
    print("Not a digit")
    


OUTPUT:
Enter the digit from 0 to 4:4
Entered digit id:Four
